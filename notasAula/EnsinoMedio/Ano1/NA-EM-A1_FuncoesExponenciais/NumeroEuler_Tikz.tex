    \begin{tikzpicture}
        % Eixos
        \draw[->,thick] (-3.5,0) -- (3.5,0) node[right] {\small \(x\)};
        \draw[->,thick] (0,-0.5) -- (0,8) node[right] {\small \(y\)};

        % Curva y = 3^x
    \draw[domain=-2.8:1.8, smooth, samples=100, blue, thick] 
        plot (\x, {3^\x}) node[above] {\(a^{x}\)};

        % Reta tangente em (0,1): y = 1 + ln(3)*x
        \draw[domain=-1.5:3, smooth, samples=2, red, thick]
            plot (\x, {1 + ln(3)*\x}) node[above] {\small Tangente};

        % Ponto (0,1)
        \filldraw (0,1) circle (2pt) node[left] {\small \((0,1)\)};
    \end{tikzpicture}
